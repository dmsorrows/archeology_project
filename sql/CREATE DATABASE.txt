CREATE TABLE Provenience (
  ProvenienceId INTEGER NOT NULL,
  ProjectNumber VARCHAR(10) NOT NULL,
  SiteNumber VARCHAR(7) NOT NULL,
  AccessionNumber VARCHAR(10) NOT NULL,
  FieldSerialNumber INTEGER NOT NULL,
  UnitNumber INTEGER NOT NULL,
  Depth VARCHAR(7) NOT NULL,
  ExcavationDate DATETIME NOT NULL,
  CONSTRAINT Provenience_PK PRIMARY KEY (ProvenienceId)
);

CREATE TABLE ArtifactData (
  ArtifactId INTEGER IDENTITY(1,1) NOT NULL,
  PeriodName VARCHAR(12) NOT NULL,
  Level1Id INTEGER NOT NULL,
  Level2Id INTEGER NOT NULL,
  Level3Id INTEGER NOT NULL,
  Level4Id INTEGER,
  AdditionalDescription VARCHAR(1000) NOT NULL,
  ArtifactCount INTEGER NOT NULL,
  ArtifactWeight DECIMAL (8,3) NOT NULL, 
  LabTechInitials VARCHAR(3) NOT NULL,
  DateAnalyzed DATETIME NOT NULL,
  ProvenienceId INTEGER NOT NULL, 
  CONSTRAINT ArtifactData_PK PRIMARY KEY (ArtifactId)
);
